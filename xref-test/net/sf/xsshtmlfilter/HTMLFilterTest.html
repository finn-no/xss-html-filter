57      @Test
58      public void testBalancingAngleBrackets() {
59          if (vFilter.isAlwaysMakeTags()) {
60              assertThat(vFilter.filter("<img src=\"foo\""), is("<img src=\"foo\" />"));
61              assertThat(vFilter.filter("i>"), is(""));
62              assertThat(vFilter.filter("<img src=\"foo\"/"), is("<img src=\"foo\" />"));
63              assertThat(vFilter.filter(">"), is(""));
64              assertThat(vFilter.filter("foo<b"), is("foo"));
65              assertThat(vFilter.filter("b>foo"), is("<b>foo</b>"));
66              assertThat(vFilter.filter("><b"), is(""));
67              assertThat(vFilter.filter("b><"), is(""));
68              assertThat(vFilter.filter("><b>"), is(""));
69          } else {
70              assertThat(vFilter.filter("<img src=\"foo\""), is("&lt;img src=\"foo\""));
71              assertThat(vFilter.filter("b>"), is("b&gt;"));
72              assertThat(vFilter.filter("<img src=\"foo\"/"), is("&lt;img src=\"foo\"/"));
73              assertThat(vFilter.filter(">"), is("&gt;"));
74              assertThat(vFilter.filter("foo<b"), is("foo&lt;b"));
75              assertThat(vFilter.filter("b>foo"), is("b&gt;foo"));
76              assertThat(vFilter.filter("><b"), is("&gt;&lt;b"));
77              assertThat(vFilter.filter("b><"), is("b&gt;&lt;"));
